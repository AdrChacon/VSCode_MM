100  // std::istream/std::streambuf use std::char_traits<char>::to_int_type, to
101  // ensure that std::char_traits<char>::eof() and the character 0xFF do not
102  // end up as the same value, eg. 0xFFFFFFFF.
103  std::char_traits<char>::int_type get_character()
104  {
105  auto res = sb->sbumpc();
106  // set eof manually, as we don't use the istream interface.
107  if (JSON_HEDLEY_UNLIKELY(res == EOF))
108  {
109  is->clear(is->rdstate() | std::ios::eofbit);
110  }
111  return res;
112  }

116  std::istream* is = nullptr;
117  std::streambuf* sb = nullptr;

176  utf8_bytes[0] = std::char_traits<char>::eof();

178  }
179  else

182  const auto wc = input.get_character();
183 
184  // UTF-32 to UTF-8 encoding

186  {
187  utf8_bytes[0] = static_cast<std::char_traits<char>::int_type>(wc);
188  utf8_bytes_filled = 1;

193  utf8_bytes[1] = static_cast<std::char_traits<char>::int_type>(0x80u | (static_cast<unsigned int>(wc) & 0x3Fu));
194  utf8_bytes_filled = 2;
195  }
196  else if (wc <= 0xFFFF)
197  {
198  utf8_bytes[0] = static_cast<std::char_traits<char>::int_type>(0xE0u | ((static_cast<unsigned int>(wc) >> 12u) & 0x0Fu));
199  utf8_bytes[1] = static_cast<std::char_traits<char>::int_type>(0x80u | ((static_cast<unsigned int>(wc) >> 6u) & 0x3Fu));
200  utf8_bytes[2] = static_cast<std::char_traits<char>::int_type>(0x80u | (static_cast<unsigned int>(wc) & 0x3Fu));
201  utf8_bytes_filled = 3;
202  }
203  else if (wc <= 0x10FFFF)
204  {
205  utf8_bytes[0] = static_cast<std::char_traits<char>::int_type>(0xF0u | ((static_cast<unsigned int>(wc) >> 18u) & 0x07u));
206  utf8_bytes[1] = static_cast<std::char_traits<char>::int_type>(0x80u | ((static_cast<unsigned int>(wc) >> 12u) & 0x3Fu));
207  utf8_bytes[2] = static_cast<std::char_traits<char>::int_type>(0x80u | ((static_cast<unsigned int>(wc) >> 6u) & 0x3Fu));
208  utf8_bytes[3] = static_cast<std::char_traits<char>::int_type>(0x80u | (static_cast<unsigned int>(wc) & 0x3Fu));
209  utf8_bytes_filled = 4;
210  }
211  else
212  {

294  {
