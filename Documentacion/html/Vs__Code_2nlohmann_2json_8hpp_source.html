1118  void destroy(value_t t) noexcept
1119  {
1120  // flatten the current json_value to a heap-allocated stack
1121  std::vector<basic_json> stack;
1122 
1123  // move the top-level items to stack
1124  if (t == value_t::array)
1125  {
1126  stack.reserve(array->size());
1127  std::move(array->begin(), array->end(), std::back_inserter(stack));
1128  }
1129  else if (t == value_t::object)
1130  {
1131  stack.reserve(object->size());
1132  for (auto&& it : *object)
1133  {
1134  stack.push_back(std::move(it.second));
1135  }
1136  }
1137 
1138  while (not stack.empty())
1139  {
1140  // move the last item to local variable to be processed
1141  basic_json current_item(std::move(stack.back()));
1142  stack.pop_back();
1143 
1144  // if current_item is array/object, move
1145  // its children to the stack to be processed later
1146  if (current_item.is_array())
1147  {
1148  std::move(current_item.m_value.array->begin(), current_item.m_value.array->end(),
1149  std::back_inserter(stack));
1150 
1151  current_item.m_value.array->clear();
1152  }
1153  else if (current_item.is_object())
1154  {
1155  for (auto&& it : *current_item.m_value.object)
1156  {
1157  stack.push_back(std::move(it.second));
1158  }
1159 
1160  current_item.m_value.object->clear();
1161  }
1162 
1163  // it's now safe that current_item get destructed
1164  // since it doesn't have any children
1165  }
1166 
1167  switch (t)
1168  {
1169  case value_t::object:
1170  {
1171  AllocatorType<object_t> alloc;
1172  std::allocator_traits<decltype(alloc)>::destroy(alloc, object);
1173  std::allocator_traits<decltype(alloc)>::deallocate(alloc, object, 1);
1174  break;
1175  }
1176 
1177  case value_t::array:
1178  {
1179  AllocatorType<array_t> alloc;
1180  std::allocator_traits<decltype(alloc)>::destroy(alloc, array);
1181  std::allocator_traits<decltype(alloc)>::deallocate(alloc, array, 1);

1595  basic_json(initializer_list_t init,
1596  bool type_deduction = true,
1597  value_t manual_type = value_t::array)
1598  {
1599  // check if each element is an array with two elements whose first
1600  // element is a string
1601  bool is_an_object = std::all_of(init.begin(), init.end(),
1602  [](const detail::json_ref<basic_json>& element_ref)
1603  {
1604  return element_ref->is_array() and element_ref->size() == 2 and (*element_ref)[0].is_string();
1605  });
1606 
1607  // adjust type if type deduction is not wanted
1608  if (not type_deduction)
1609  {
1610  // if array is wanted, do not create an object though possible
1611  if (manual_type == value_t::array)
1612  {
1613  is_an_object = false;
1614  }
1615 
1616  // if object is wanted but impossible, throw an exception
1617  if (JSON_HEDLEY_UNLIKELY(manual_type == value_t::object and not is_an_object))
1618  {
1619  JSON_THROW(type_error::create(301, "cannot create object from initializer list"));
1620  }
1621  }
1622 
1623  if (is_an_object)
1624  {
1625  // the initializer list is a list of pairs -> create object
1626  m_type = value_t::object;
1627  m_value = value_t::object;
1628

8105  {
8106  // make a working copy to apply the patch to

8108 
8109  // the valid JSON Patch operations
8110  enum class patch_operations {add, remove, replace, move, copy, test, invalid};
8111 
8112  const auto get_op = [](const std::string & op)
8113  {
8114  if (op == "add")
8115  {
8116  return patch_operations::add;
8117  }
8118  if (op == "remove")
8119  {
8120  return patch_operations::remove;
8121  }
8122  if (op == "replace")
8123  {
8124  return patch_operations::replace;
8125  }
8126  if (op == "move")
8127  {
8128  return patch_operations::move;
8129  }
8130  if (op == "copy")
8131  {
8132  return patch_operations::copy;
8133  }
8134  if (op == "test")
8135  {
8136  return patch_operations::test;
8137  }
8138 
8139  return patch_operations::invalid;
8140  };
8141 
8142  // wrapper for "add" operation; add value at ptr
8143  const auto operation_add = [&result](json_pointer & ptr, basic_json val)
8144  {
8145  // adding to the root of the target document means replacing it
8146  if (ptr.empty())
8147  {
8148  result = val;
8149  return;
8150  }
8151 
8152  // make sure the top element of the pointer exists
8153  json_pointer top_pointer = ptr.top();
8154  if (top_pointer != ptr)
8155  {
8156  result.at(top_pointer);
8157  }
8158 
8159  // get reference to parent of JSON pointer ptr
8160  const auto last_path = ptr.back();
8161  ptr.pop_back();
8162  basic_json& parent = result[ptr];
8163 
8164  switch (parent.m_type)
8165  {
8166  case value_t::null:
8167  case value_t::object:
8168  {
8169  // use operator[] to add value
8170  parent[last_path] = val;
8171  break;
8172  }
8173 
8174  case value_t::array:
8175  {
8176  if (last_path == "-")
8177  {
8178  // special case: append to back
8179  parent.push_back(val);
8180  }
8181  else
8182  {
8183  const auto idx = json_pointer::array_index(last_path);
8184  if (JSON_HEDLEY_UNLIKELY(static_cast<size_type>(idx) > parent.size()))
8185  {
8186  // avoid undefined behavior
8187  JSON_THROW(out_of_range::create(401, "array index " + std::to_string(idx) + " is out of range"));
8188  }
8189 
8190  // default case: insert add offset
8191  parent.insert(parent.begin() + static_cast<difference_type>(idx), val);
8192  }
8193  break;
8194  }
8195 
8196  // if there exists a parent it cannot be primitive
8197  default: // LCOV_EXCL_LINE
8198  assert(false); // LCOV_EXCL_LINE
8199  }
8200  };

8202  // wrapper for "remove" operation; remove value at ptr
8203  const auto operation_remove = [&result](json_pointer & ptr)
8204  {
8205  // get reference to parent of JSON pointer ptr
8206  const auto last_path = ptr.back();
8207  ptr.pop_back();
8208  basic_json& parent = result.at(ptr);
8209 
8210  // remove child
8211  if (parent.is_object())
8212  {
8213  // perform range check
8214  auto it = parent.find(last_path);
8215  if (JSON_HEDLEY_LIKELY(it != parent.end()))
